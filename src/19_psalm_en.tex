\psalmres{Responsorial Psalm}{Ps 105 (104), 1-2. 3-4. 6-7. 8-9}{Rejoice, O hearts that seek the Lord.\\\indent\red{or}: Alleluia.}

\psalmverse{1}{Give thanks to the LORD, invoke his name;\redast\\make known among the nations his deeds.}
\psalmverse{2}{Sing to him, sing his praise,\redast\\proclaim all his wondrous deeds.}
\indent\rubric{Ref.}
\psalmverse{3}{Glory in his holy name;\redast\\rejoice, O hearts that seek the LORD!}
\psalmverse{4}{Look to the LORD in his strength;\redast\\seek to serve him constantly.}
\indent\rubric{Ref.}
\psalmverse{5}{You descendants of Abraham, his servants,\redast\\sons of Jacob, his chosen ones!}
\psalmverse{6}{He, the LORD, is our God;\redast\\throughout the earth his judgments prevail.}
\indent\rubric{Ref.}
\psalmverse{7}{He remembers forever his covenant\redast\\which he made binding for a thousand generations}
\psalmverse{8}{Which he entered into with Abraham\redast\\and by his oath to Isaac.}
